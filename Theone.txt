#include<iostream>
#include<conio.h>
#include<string.h>
#include<stdlib.h>
 
using namespace std; 

int MenuChoose1, s;

struct vector{
		long *pointer;
		int size;
		};
		
vector V,tmp;	

void init(vector *vr, int size){
	vr->pointer=new long[size];
	vr->size=size;
	cout<<"Enter elements of Vector."<<endl;
	for(int i=0;i<size;i++)
		cin>>vr->pointer[i];	
}

void DLT(vector *vr){
	delete[]vr->pointer;
	vr->size=0;
}

void change(vector *vr, int s){
	tmp.pointer=new long[vr->size];
	for(int i=0;i<vr->size;i++)
		tmp.pointer[i]=vr->pointer[i];
	tmp.size=vr->size;
	delete[]vr->pointer;
	vr->pointer=new long[s];
	vr->size=s;
	if(vr->size>tmp.size){
		for(int i=0;i<tmp.size;i++)
			vr->pointer[i]=tmp.pointer[i];
		cout<<"Add "<<vr->size-tmp.size<<" more elements."<<endl;
		for(int i=tmp.size; i < vr->size; i++)
			cin>>vr->pointer[i];
	}
	else{
		for(int i=0;i<vr->size;i++)
			vr->pointer[i]=tmp.pointer[i];
	}
	DLT(&tmp);
}

int max(int a, int b){
	if(a>b)
		return a;
	else
		return b;
}
int sum(){
	int res=0;
	for(int i=0;i<s;i++){
		if(V.pointer[i]%2==0)
			res=res+V.pointer[i];
	}
	return res;
}

	while(1){
		cout<<"1.Input data."<<endl;

		cout<<"3.Set/change Vector's size."<<endl;

		cout<<"5.Calculating the sum of the even elements of a Vector."<<endl;
		cout<<"6.Exit."<<endl;
		cin>>MenuChoose1;
		system("cls");
		switch(MenuChoose1){
			case 1 :{
				cout<<"How many elements would you like to enter?"<<endl;
				cin>>s;
				system("cls");
				init (&V,s);
				system("pause");
				system("cls");
				break;
			}
			case 2 :{
					cout<<"Vector :";
					for(int i=0;i<s;i++)
						cout<<V.pointer[i]<<" ";
					cout<<endl;
					system("pause");
					system("cls");
				break;
			}
			case 3 :{
				cout<<"Novii razmer:";
				cin>>s;
				change(&V,s);
				cout<<"Size Changed."<<endl;
				system("pause");
				system("cls");
				break;
			}
			case 4 :{
				int mr, ms, pos;
				mr=V.pointer[0];
				pos=1;
				for(int i=1;i<s;i++){
					ms=max(mr,V.pointer[i]);
					if(ms!=mr){
						pos=i+1;
						mr=ms;
					}					
				}
				cout<<"Nomer maximalinih elementov vectora : "<<pos<<".\nMaximum is :"<<mr<<"."<<endl;
				system("pause");

			case 5 : {
